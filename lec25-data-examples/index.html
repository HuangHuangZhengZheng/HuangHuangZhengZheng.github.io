def digit_dict(s):
    """
    >>> digit_dict([5, 2, 13])
    {2: [2], 3: [13], 5: [5]}
    """
    last_digit = [x % 10 for x in s]
    return {d: [x for x in s if x % 10 == d] for d in range(10) if d in last_digit}
