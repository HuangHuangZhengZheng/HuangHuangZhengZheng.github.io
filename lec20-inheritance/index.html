class A:
    def __init__(self, x):
        self.x = x

class B:
    def __init__(self, y):
        self.y = y

    def __str__(self):
        return f"B(y={self.y})"

class C(A, B):
    def __init__(self, x, y):
        super().__init__(x)
        self.y = y

    def __str__(self):
        return f"C(x={self.x}, y={self.y})"

    def f(self):
        return self.x + self.y

c = C(1, 2)
print(c)
print(c.f())
