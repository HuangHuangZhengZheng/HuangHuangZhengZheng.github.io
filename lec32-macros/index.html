        ()

def trace(fn):
    def traced(n):
        print(f"Calling {fn.__name__} with {n}")
        return fn(n) # n is fn's argument!
    return traced

@trace
def fact(n):
    if n == 0:
        return 1
    else:
        return n * fact(n-1)

print(fact(5))
